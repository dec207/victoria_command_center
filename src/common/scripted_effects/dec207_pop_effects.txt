﻿# VCC Scripted Effects
# Prefix: dec207_

# 1. Individual Pop Addition (Safe Method)
dec207_add_pop_100k_effect = {
	# Victoria 3 create_pop logic is sensitive to culture references.
	# The Wiki and Vanilla files confirm culture keys must match internal IDs.
	# We use a conditional check to ensure the correct culture is applied.
	if = {
		limit = { 
			owner = { country_has_primary_culture = cu:korean } 
		}
		create_pop = {
			culture = korean
			size = 100000
		}
	}

			owner = { country_has_primary_culture = cu:british } 
		}
		create_pop = {
			culture = british
			size = 100000
		}
	}
	else_if = {
		limit = { 
			owner = { country_has_primary_culture = cu:han } 
		}
		create_pop = {
			culture = han
			size = 100000
		}
	}
	else_if = {
		limit = { 
			owner = { country_has_primary_culture = cu:japanese } 
		}
		create_pop = {
			culture = japanese
			size = 100000
		}

			culture = yankee
			size = 100000
		}
	}
	# Fallback for ANY state: Add 100k to existing pops proportionally
	else = {
		every_scope_pop = {
			add_pop_size = 100000
		}
	}
}
