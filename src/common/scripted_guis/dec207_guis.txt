# 1. Gold Addition (Verified Working)
dec207_add_gold_1M_gui = {
	scope = country
	effect = {
		add_treasury = 1000000
	}
}

# 2. Population Addition (Fixed Logic)
# Using 'every_scope_state' or 'any_scope_state' check is safer, 
# but for a simple cheat, we target all states or just capital.
# Let's ensure 'owner' and 'primary_culture' are correctly scoped.
dec207_add_pop_100k_gui = {
	scope = country
	effect = {
		capital_scope = {
			create_pop = {
				culture = owner.primary_culture
				size = 100000
			}
		}
	}
}

# 3. Infamy Reset (Fixed Logic)
# 'change_infamy' is the correct effect for Victoria 3.
dec207_reset_infamy_gui = {
	scope = country
	effect = {
		change_infamy = -1000
	}
}
