# VCC Dashboard Scripted GUI
# Prefix: dec207_

dec207_add_gold_1M_gui = {
	scope = country
	effect = {
		add_treasury = 1000000
	}
}

dec207_add_pop_100k_gui = {
	scope = country
	effect = {
		capital_scope = {
			create_pop = {
				culture = owner.primary_culture
				size = 100000
			}
		}
	}
}

dec207_reset_infamy_gui = {
	scope = country
	effect = {
		# In Victoria 3, 'set_infamy' is not a standard effect.
		# We must use 'change_infamy' with a negative value equal to current infamy.
		change_infamy = -1000
	}
}
